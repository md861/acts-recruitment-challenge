112 def replay_hash(config: ModelConfig, ticks: int) -> str:
113 
114  model = PopulationModel(config)
115  digest = hashlib.sha256()
116  for _ in range(max(1, ticks)):
117  model.step()
118  for agent in sorted(model.agents, key=lambda item: item.id):
119  digest.update(
120  (
121  f"{model.tick}|{agent.id}|{agent.role}|{agent.status}|"
122  f"{agent.position.x}|{agent.position.y}|"
123  f"{agent.heading.dx}|{agent.heading.dy};"
124  ).encode("utf-8")
125  )
126  return digest.hexdigest()
