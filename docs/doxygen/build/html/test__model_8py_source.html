234  def _movement_into(self, terrain, target_type):
235  for index, candidate in enumerate(terrain.cell_types):
236  if candidate != target_type:
237  continue
238  target = (index % terrain.width, index // terrain.width)
239  for dx, dy in ((1, 0), (-1, 0), (0, 1), (0, -1)):
240  source = (target[0] - dx, target[1] - dy)
241  if source[0] < 0 or source[1] < 0:
242  continue
243  if source[0] >= terrain.width or source[1] >= terrain.height:
244  continue
245  if terrain.cell_type_at(*source) == CellType.NORMAL:
246  return source, target, (dx, dy)
247  self.fail(f"No normal source found adjacent to {target_type.value}")
