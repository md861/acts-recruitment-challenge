383 def _read_png_rgb(path: Path) -> tuple[int, int, list[tuple[int, int, int]]]:
384  data = path.read_bytes()
385  if data[:8] != b"\x89PNG\r\n\x1a\n":

388  width = height = bit_depth = color_type = interlace = None

391  while pos < len(data):
392  length = struct.unpack(">I", data[pos : pos + 4])[0]
393  chunk_type = data[pos + 4 : pos + 8]
394  chunk = data[pos + 8 : pos + 8 + length]
395  pos += length + 12
396 
397  if chunk_type == b"IHDR":

400  height,
401  bit_depth,
402  color_type,
403  _compression,
404  _filter,
405  interlace,

434 ) -> list[bytearray]:
435  row_size = width * channels
436  previous = bytearray(row_size)

439

460 def _png_filter_predictor(
461  filter_type: int, left: int, up: int, upper_left: int
462 ) -> int:
463  if filter_type == 0:
464  return 0
465  if filter_type == 1:
466  return left
467  if filter_type == 2:
468  return up
469  if filter_type == 3:
470  return (left + up) // 2
471  if filter_type == 4:
472  estimate = left + up - upper_left
473  left_distance = abs(estimate - left)
474  up_distance = abs(estimate - up)
475  upper_left_distance = abs(estimate - upper_left)
476  if left_distance <= up_distance and left_distance <= upper_left_distance:
477  return left
478  if up_distance <= upper_left_distance:
479  return up
480  return upper_left
481  raise ValueError(f"Unsupported PNG filter type: {filter_type}")
